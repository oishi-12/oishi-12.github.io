### Codes of Unsupervised classification of Hathazari upazilla
-----
```Go
var table = ee.FeatureCollection("projects/poetic-flight-476905-g6/assets/bgd_admbnda_adm3_bbs_20201113");
var hathazari = table.filter(ee.Filter.eq("ADM3_EN", "Hathazari"))
print(hathazari)

var sentinel2 = ee.ImageCollection("COPERNICUS/S2_SR_HARMONIZED")
                .filterBounds(hathazari)
                .filterDate('2023-01-01', '2023-12-31')
                .filter(ee.Filter.lt('CLOUDY_PIXEL_PERCENTAGE',20))
                .median()
                .clip(hathazari);
var bands = ['B2','B3','B4','B8'];
var image = sentinel2.select(bands)
Map.centerObject(hathazari,10);
Map.addLayer(image,{bands: ['B4', 'B3', 'B2'],
                    min: 0,
                    max: 3000},'Sentinel_2_RGB');
var training = image.sample({
  region: hathazari,
  scale: 10,
  numPixels: 5000
});        
print(training)
var clusterer =ee.Clusterer.wekaKMeans(4).train(training);
var result = image.cluster(clusterer);
print(result)

Map.addLayer(result.randomVisualizer(), {}, 'K-means Clusters');
Export.image.toDrive({
  image: result,
  description: 'KMeans_clusters',
  scale: 10,
  region: hathazari,
  maxPixels: 1e13
});
```
